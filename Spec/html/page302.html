   ((puzzle-array :initform (make-array '(4 4))))
   (:menu-bar t)
   (:panes
     (display
       (outlining ()
         (make-pane 'application-pane
                    :text-cursor nil
                    :width :compute
                    :height :compute
                    :incremental-redisplay T
                    :display-function 'draw-puzzle))))
   (:layout
     (:default display)))
